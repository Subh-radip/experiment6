class Admin:
    def _init_(self, name):
        self.name = name
        self.facilities = {}

    def add_facility(self, facility_name):
        if facility_name in self.facilities:
            print(f"{facility_name} already exists.")
        else:
            self.facilities[facility_name] = []
            print(f"{facility_name} added successfully.")

    def remove_facility(self, facility_name):
        if facility_name in self.facilities:
            del self.facilities[facility_name]
            print(f"{facility_name} removed successfully.")
        else:
            print(f"{facility_name} not found.")
